        INSERT INTO estimates (id, filename, data, data_version, created_at, updated_at)
        VALUES (?, ?, ?, ?, ?, ?)

    db.prepare(`
        INSERT INTO backups (id, estimate_id, data, created_at)
        VALUES (?, ?, ?, ?)
    `).run(id, estimateId, dataStr, now);
}
