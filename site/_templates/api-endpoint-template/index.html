app.[method]('/api/[resource]/[path]', async (req, res) => {
    try {
        const { field1, field2 } = req.body;

        // Validation
        if (!field1) {
            return res.status(400).json({
                success: false,
                error: 'field1 is required'
            });
        }

        // Process request
        const result = await storage.someMethod(field1, field2);

        // Success response
        res.json({
            success: true,
            data: result
        });
    } catch (error) {
        res.status(500).json({
            success: false,
            error: error.message
        });
    }
});
